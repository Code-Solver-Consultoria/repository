1   package br.com.codesolver.dcounter3d.security;
2   
3   import java.security.Principal;
4   import java.util.logging.Level;
5   import java.util.logging.Logger;
6   
7   import org.wildfly.extension.elytron.capabilities.PrincipalTransformer;
8   
9   /**
10   * Tratamento para o retorno corrreto da identificação do usuário.
11   *
12   * @author <a href="mailto:luciano@codesolver.com.br">Luciano Vieira Rodrigues</a>
13   * @since 2025-08-25
14   */
15  public class CSPrincipalTransformer implements PrincipalTransformer {
16  
17  	/**
18  	 * Log da classe.
19  	 */
20  	private static final Logger LOGGER = Logger.getLogger(CSPrincipalTransformer.class.getName());
21  	
22  	@Override
23  	public Principal apply(Principal principal) {
24  		if (LOGGER.isLoggable(Level.FINEST)) {
25  			LOGGER.log(Level.FINEST, String.format("Tranformando a identidade do usuário \"%s\".", 
26  				principal.getName()));
27  		}
28  		CSPrincipal result = null;
29  		CSPrincipalCache cache = CSPrincipalCache.getInstance();
30  		if (cache.constains(principal.getName())) {
31  			result = cache.get(principal.getName()); 
32  		} else {
33  			result = new CSPrincipal(principal.getName());
34  			cache.add(result);
35  		}
36  		return result;
37  	}
38  }

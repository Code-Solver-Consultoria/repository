106     public CSRealmIdentity(Principal principal, CSConfiguration configuration) {
107         if (LOGGER.isLoggable(Level.FINER)) {
108             LOGGER.log(Level.FINER, String.format("Preparando a autenticação para o usuário \"%s\".", 
109                 principal.getName()));
110         }
111         this.configuration	= configuration;
112         this.cache 			= CSPrincipalCache.getInstance();
113         if (cache.constains(principal.getName())) {
114             this.principal	= cache.get(principal.getName());	
115         } else {
116             this.principal	= new CSPrincipal(principal.getName());
117             this.cache.add(this.principal);
118         }
119         this.attributes		= new MapAttributes();
120         this.uuid			= new UUIDGenerator(configuration.getUUIDNode());
121     }

167     @Override
168     @SuppressWarnings({
169         "ReturnCount", "CyclomaticComplexity", "JavaNCSS", "NestedIfDepth",
170         "NPathComplexity", "ExecutableStatementCount", "MethodLength"
171     })
172     public boolean verifyEvidence(Evidence evidence) throws RealmUnavailableException {
173         if (LOGGER.isLoggable(Level.FINEST)) {
174             LOGGER.log(Level.FINEST, String.format("Validando a evidência de identificação do usuário %s", 
175                 principal.getName()));
176         }
177         if (evidence instanceof PasswordGuessEvidence) {
178             PasswordGuessEvidence password = (PasswordGuessEvidence) evidence;
179             try (Database database = new Database(configuration)) {
180                 Usuario usuario = findUsuario(database, principal.getName());
181                 if (usuario == null) {
182                     if (LOGGER.isLoggable(Level.FINEST)) {
183                         LOGGER.log(Level.FINEST, String.format(INVALID_USER, principal.getName()));
184                     }
185                     return false;
186                 }
187                 if (!usuario.getAtivo()) {
188                     LOGGER.warning(String.format("Usuário \"%s\" está inativo.", principal.getName()));
189                     return false;
190                 }
191                 Senha senha = selectSenha(database, usuario);
192                 if (senha == null) {
193                     LOGGER.warning(String.format("Usuario \"%s\" não possui uma senha definida.", principal.getName()));
194                     return false;
195                 }
196                 // Se a senha não passar nos testes de complexidade...
197                 if (usuario.getTipo() != UsuarioTipo.INTERNO && !passwordComplexityIsValid(database, usuario, senha)) {
198                     LOGGER.warning(String.format("Senha do usuário \"%s\" não passou nos testes de complexidade.", 
199                         principal.getName()));
200                     return false;
201                 }
202                 // Compara as senhas:
203                 CharBuffer chars 	= CharBuffer.wrap(password.getGuess());
204                 ByteBuffer buffer 	= Charset.defaultCharset().encode(chars);
205                 byte[] bytes 		= Arrays.copyOfRange(buffer.array(), buffer.position(), buffer.limit());
206                 Digester digester	= DigesterFactory.getInstance(AlgorithmType.SHA_512);
207                 digester.update(bytes);
208                 char[] digested		= digester.digest().toCharArray();
209                 boolean valid		= Arrays.equals(digested, senha.getValor().toCharArray());
210                 if (LOGGER.isLoggable(Level.FINEST)) {
211                     if (valid) {
212                         LOGGER.log(Level.FINEST, String.format("Usuário \"%s\" autenticado.", principal.getName()));
213                     } else {
214                         LOGGER.log(Level.FINEST, String.format(INVALID_USER, principal.getName()));
215                     }
216                 }
217                 // Se a senha é válida e não está expirada:
218                 if (valid) {
219                     // Reiniciando a contagem de erros e registra o momento de acesso da senha:
220                     database.beginTransaction();
221                     senha.setTentativas(0);
222                     senha.setAcesso(LocalDateTime.now());
223                     database.merge(senha);
224                     database.commit();
225                     // Informações adicionais do usuaŕio:
226                     principal.setID(usuario.getId());
227                     principal.setFullName(usuario.getNome());
228                     principal.setExpired(senha.getExpirado());
229                     principal.setOperator(usuario.getTipo() == UsuarioTipo.OPERADOR);
230                     cache.add(principal);
231                     // Arquivando as permissões, se a senha não estiver expirada:
232                     if (!principal.isExpired()) {
233                         List<Perfil> perfis = usuario.getPerfis();
234                         for (Perfil perfil : perfis) {
235                             if (perfil.getAtivo()) {
236                                 perfil.getFuncionalidades().forEach(f -> processFuncionalidade(f));
237                             }
238                         }
239                         usuario.getFuncionalidades().forEach(f -> processFuncionalidade(f));
240                     }
241                     // Se o usuário for um operador de sistema, inicia a sessão de auditoria:
242                     if (principal.isOperator()) {
243                         database.beginTransaction();
244                         Sessao sessao = new Sessao();
245                         sessao.setId(uuid.generate());
246                         sessao.setUsuario(usuario);
247                         sessao.setAbertura(LocalDateTime.now());
248                         sessao.setIp(getRemoteIP());
249                         database.persist(sessao);
250                         database.commit();
251                         principal.setSessionID(sessao.getId());
252                     }
253                 } else {
254                     // Incrementando a quantidade de erros de acesso:
255                     database.beginTransaction();
256                     senha.setTentativas(senha.getTentativas() + 1);
257                     database.merge(senha);
258                     database.commit();
259                 }
260                 return valid;
261             } catch (PersistenceException e) {
262                 LOGGER.log(Level.FINEST, String.format("Erro na camada de persistência: %s.", e.getMessage()));
263             }							// try (Database database ... (auto close)
264         }								// if (evidence instanceof PasswordGuessEvidence)
265         return false;
266     }

268     /**
269      * Aplica as regras de complexidade na senha e inativa o usuário ou expira a senha conforme necessidade.
270      *
271      * @param database Acesso a camada de persistência.
272      * @param usuario Usuário tentando autenticar o sistema.
273      * @param senha Senha que será validada.
274      * @return 
275      *     true Senha passou em todos os teste de complexidade.
276      *     false Senha não pasou em algum teste de complexidade. 
277      */
278     private boolean passwordComplexityIsValid(Database database, Usuario usuario, Senha senha) {
279         // É para aplicar as regras de complexidade de senha?
280         boolean passwordComplex = Boolean.parseBoolean(database.find(Parametro.class, 
281             Parametro.PASSWORD_COMPLEX).getValor());
282         if (passwordComplex) {
283             // Verificar atingiu o limite de erros de autenticação:
284             int retries = Integer.parseInt(database.find(Parametro.class, Parametro.PASSWORD_RETRIES).getValor());
285             if (retries > 0 && senha.getTentativas() >= retries) {
286                 LOGGER.info(String.format("Usuário \"%s\" atingiu o número máximo de tentativas de acesso com erro de senha.",
287                     usuario.getLogin()));
288                 database.beginTransaction();
289                 usuario.setAtivo(false);
290                 database.merge(usuario);
291                 database.commit();
292                 return false;
293             }
294             // Verifica se atingiu o tempo máximo de inatividade (ignorar o primeiro acesso):
295             int inactivity = Integer.parseInt(database.find(Parametro.class, Parametro.PASSWORD_INACTIVITY).getValor());
296             if (inactivity > 0 && senha.getAcesso() != null) {
297                 LocalDate limit = LocalDate.now();
298                 limit = limit.minusDays(inactivity);
299                 if (senha.getAcesso().isBefore(limit.atStartOfDay())) {
300                     LOGGER.info(String.format("Usuário \"%s\" atingiu o limite de inatividade.", usuario.getLogin()));
301                     database.beginTransaction();
302                     usuario.setAtivo(false);
303                     database.merge(usuario);
304                     database.commit();
305                     return false;
306                 }
307             }
308             // Verifica se a senha atingiu o tempo máximo de uso.
309             int lifetime = Integer.parseInt(database.find(Parametro.class, Parametro.PASSWORD_LIFETIME).getValor());
310             if (lifetime > 0) {
311                 LocalDate limit = LocalDate.now();
312                 limit = limit.minusDays(lifetime);
313                 if (senha.getData().isBefore(limit.atStartOfDay())) {
314                     LOGGER.info(String.format("Senha do Usuário \"%s\" atingiu o tempo máximo de uso.", usuario.getLogin()));
315                     database.beginTransaction();
316                     senha.setExpirado(true);
317                     database.merge(senha);
318                     database.commit();
319                     // O usuário é válido e autenticado, mas precisa alterar a senha para prosseguir...
320                 }
321             }
322         }
323         // Se passou em todos os testes:
324         return true;
325     }

327     /**
328      * Seleciona a senha ativa e validade.
329      *
330      * @param database Acesso ao módulo de persistência.
331      * @param usuario Usuário autenticado.
332      * @return Senha
333      */
334     private Senha selectSenha(Database database, Usuario usuario) {
335         Senha result = null;
336         // Recupera as senhas ativas:
337         List<Senha> senhas = usuario.getSenhas().stream()
338                 .filter(s -> s.getAtivo())
339                 .collect(Collectors.toList());
340         // Se encontrar mais que uma senha ativa, deve-se selecionar a mais nova e inativar as demais.
341         // Na prática, não deverá ocorrer nunca
342         if (senhas.size() > 1) {
343             LOGGER.log(Level.SEVERE, String.format("Usuário \"%s\" com %d senhas ativas?", usuario.getLogin(), senhas.size()));
344             List<Senha> sorted = senhas.stream()
345                     .sorted(Comparator.comparing(Senha::getData).reversed())
346                     .collect(Collectors.toList());
347             // Selecionando o registro mais novo e inativando os demais:
348             for (Senha senha : sorted) {
349                 if (result == null) {
350                     result = senha;
351                     continue;		// Retorna ao início do laço para inativar as demais senhas.
352                 }
353                 database.beginTransaction();
354                 senha.setAtivo(false);
355                 database.merge(senha);
356                 database.commit();
357             }
358         }
359         // Se ainda não possui um senha selecionada...
360         if (result == null && !senhas.isEmpty()) {
361             result = senhas.get(0);
362         }
363         return result;
364     }

366     /**
367      * Localiza um usuário por seu login.
368      *
369      * @param database Conexão com o banco de dados.
370      * @param name Nome de login do usuário.
371      * @return Usuario
372      */
373     private Usuario findUsuario(Database database, String name) {
374         // Preparando a consulta:
375         CriteriaBuilder builder			= database.getCriteriaBuilder();
376         CriteriaQuery<Usuario> criteria	= builder.createQuery(Usuario.class);
377         Root<Usuario> root				= criteria.from(Usuario.class);
378         criteria.select(root);
379         // Preparando o parâmetro de filtro por login:
380         ParameterExpression<String> login	= builder.parameter(String.class);
381         Predicate predicate					= builder.and(builder.equal(root.get(Usuario_.login), login));
382         criteria.where(predicate);
383         // Compilando a consulta:
384         TypedQuery<Usuario> query = database.findByCriteria(criteria);
385         query.setParameter(login, name);
386         return query.getSingleResult();
387     }

389     /**
390      * Adiciona as regras de acesso do usuário.
391      *
392      * @param funcionalidade Funcionalidade de autorização do sistema.
393      */
394     private void processFuncionalidade(Funcionalidade funcionalidade) {
395         if (funcionalidade.getAtivo() && funcionalidade.getModulo().getAtivo()) {
396             if (attributes.containsKey(ROLES_IDENTITY)) {
397                 attributes.get(ROLES_IDENTITY).add(funcionalidade.getId());
398             } else {
399                 attributes.addFirst(ROLES_IDENTITY, funcionalidade.getId());
400             }
401         }
402     }

418     /**
419      * Obtém o endereço IP do usuário remoto.
420      *
421      * <p>
422      * Este método so consegue obter o endereçp do usuário quando a conexão remote
423      * for através de http.
424      *
425      * @return retorna o IP do usuário remoto
426      */
427     private String getRemoteIP() {
428         String result = null;
429         try {
430             ServletRequest request = (HttpServletRequest) PolicyContext.getContext(
431                     "jakarta.servlet.http.HttpServletRequest");
432             if (request != null) {
433                 result = request.getRemoteAddr();
434             }
435         } catch (IllegalArgumentException | SecurityException | PolicyContextException e) {
436             LOGGER.log(Level.WARNING, 
437                 String.format("Não foi possivel obter o IP do usuário \"%s\" remoto para registrar na auditoria",
438                     principal.getName()));
439         }
440         return result;
441     }

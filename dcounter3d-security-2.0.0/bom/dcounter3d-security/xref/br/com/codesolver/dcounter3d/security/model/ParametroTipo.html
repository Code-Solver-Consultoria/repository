103 	/**
104 	 * Recupera um Tipo de Parâmetro utilizando seu valor de persistência no banco
105 	 * de dados.
106 	 *
107 	 * @param value Valor de persistência no banco de dados.
108 	 * @return {@link ParametroTipo}.
109 	 * @throws IllegalArgumentException Valor inválido.
110 	 */
111 	public static ParametroTipo valueOf(long value) {
112 		for (ParametroTipo type : values()) {
113 			if (type.getValue() == value) {
114 				return type;
115 			}
116 		}
117 		// Se chegou nesse ponto, o valor é inválido.
118 		throw new IllegalArgumentException(String.format("Valor %d não é um tipo válido.", value));
119 	}

3   import java.security.Principal;
4   import java.security.spec.AlgorithmParameterSpec;
5   import java.util.Map;
6   import java.util.logging.Level;
7   import java.util.logging.Logger;
8   
9   import org.wildfly.security.auth.SupportLevel;
10  import org.wildfly.security.auth.server.RealmIdentity;
11  import org.wildfly.security.auth.server.RealmUnavailableException;
12  import org.wildfly.security.auth.server.SecurityRealm;
13  import org.wildfly.security.credential.Credential;
14  import org.wildfly.security.credential.PasswordCredential;
15  import org.wildfly.security.evidence.Evidence;
16  import org.wildfly.security.evidence.PasswordGuessEvidence;
17  import org.wildfly.security.password.interfaces.DigestPassword;
18  import org.wildfly.security.password.spec.DigestPasswordAlgorithmSpec;
19  
20  /**
21   * Verifica a autenticação dos usuários.
22   *
23   * @author <a href="mailto:luciano@codesolver.com.br">Luciano Vieira Rodrigues</a>
24   * @since 2025-08-25
25   */
26  public class CSRealm implements SecurityRealm {
27  
28  	/**
29  	 * Log da classe.
30  	 */
31  	private static final Logger LOGGER = Logger.getLogger(CSRealm.class.getName());
32  	
33  	/**
34  	 * Armazena a configuração do módulo de autenticação.
35  	 */
36  	private CSConfiguration configuration;
37  
38  	/**
39  	 * Inicializa o gerenciador de autenticação.
40  	 *
41  	 * @param config Configuração da autenticação.
42  	 */
43  	public void initialize(Map<String, String> config) {
44  		if (LOGGER.isLoggable(Level.CONFIG)) {
45  			LOGGER.log(Level.CONFIG, "Inicializando o módulo de autenticação de usuários.");
46  		}
47  		if (LOGGER.isLoggable(Level.FINEST) && !config.isEmpty()) {
48  			LOGGER.log(Level.FINEST, "Configuração:");
49  			config.forEach((k, v) -> LOGGER.log(Level.FINEST, String.format("\t%s=%s", k, v)));
50  		}
51  		this.configuration = new CSConfiguration(config);
52  	}
53  
54  	@Override
55  	@SuppressWarnings("BooleanExpressionComplexity")
56  	public SupportLevel getCredentialAcquireSupport(Class<? extends Credential> credentialType, String algorithmName,
57  			AlgorithmParameterSpec parameterSpec) throws RealmUnavailableException {
58  		if (LOGGER.isLoggable(Level.FINEST)) {
59  			LOGGER.log(Level.FINEST, 
60  				String.format("Verificando o suporte ao tipo de credencial %s com o " + 
61  					"algorítimo %s e parâmetro de especificação %s", 
62  					credentialType, algorithmName, parameterSpec));
63  		}		
64  		if (PasswordCredential.class.isAssignableFrom(credentialType) &&
65                  (algorithmName == null || algorithmName.equals(DigestPassword.ALGORITHM_DIGEST_SHA_512)) &&
66                  (parameterSpec == null || parameterSpec instanceof DigestPasswordAlgorithmSpec)) {
67  			return SupportLevel.SUPPORTED;
68  		}
69          return SupportLevel.UNSUPPORTED;
70  	}
71  
72  	@Override
73  	public SupportLevel getEvidenceVerifySupport(Class<? extends Evidence> evidenceType, String algorithmName)
74  			throws RealmUnavailableException {
75  		if (LOGGER.isLoggable(Level.FINEST)) {
76  			LOGGER.log(Level.FINEST, 
77  				String.format("Verificando o suporte ao tipo de evidência %s com o algorítimo %s.", 
78  					evidenceType, algorithmName));
79  		}		
80  		if (PasswordGuessEvidence.class.isAssignableFrom(evidenceType)) {
81  			return SupportLevel.POSSIBLY_SUPPORTED;
82  		} 
83  		return SupportLevel.UNSUPPORTED;
84  	}
85  
86  	@Override
87  	public RealmIdentity getRealmIdentity(Principal principal) throws RealmUnavailableException {
88  		if (this.configuration == null) {
89              throw new RealmUnavailableException("CSRealm não inicializado...");
90          }
91  		RealmIdentity result = null;
92  		if (principal instanceof CSPrincipal) {
93  			result = new CSRealmIdentity(principal, configuration);
94  		} else {
95  			result = RealmIdentity.NON_EXISTENT;
96  		}
97  		return result;
98  	}
99  }

6   /**
7    * Converte um Tipo de Usuário para seu valor de armazenamento no banco de dados e vice-versa.
8    *
9    * @author <a href="mailto:luciano@codesolver.com.br">Luciano Vieira Rodrigues</a>
10   * @since 2025-08-25
11   */
12  @Converter(autoApply = true)
13  public class UsuarioTipoConverter implements AttributeConverter<UsuarioTipo, Long> {
14  	
15  	/**
16  	 * Construtor padrão.
17  	 */
18  	public UsuarioTipoConverter() {
19  	}
20  	
21  	@Override
22  	public Long convertToDatabaseColumn(UsuarioTipo attribute) {
23  		return attribute.getValue();
24  	}
25  
26  	@Override
27  	public UsuarioTipo convertToEntityAttribute(Long dbData) {
28  		return UsuarioTipo.valueOf(dbData);
29  	}
30  }

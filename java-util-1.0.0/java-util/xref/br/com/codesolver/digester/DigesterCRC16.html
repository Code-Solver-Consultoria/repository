76  	@Override
77  	public String digest() {
78  		return Integer.toHexString(crc & SHORT_MASK);
79  	}

81  	/**
82  	 * Calcula o CRC com base na matriz de bytes recebida.
83  	 *
84  	 * @param buffer Matriz de bytes.
85  	 */
86  	@Override
87  	public void update(byte[] buffer) {
88  		for (int i = 0; i < buffer.length; i++) {
89  			byte b = buffer[i];
90  			int indice = crc >> Byte.SIZE ^ b;
91  			crc = crc << Byte.SIZE & SHORT_MASK ^ CRC16_LOOKUP[indice & BYTE_MASK];
92  		}
93  	}

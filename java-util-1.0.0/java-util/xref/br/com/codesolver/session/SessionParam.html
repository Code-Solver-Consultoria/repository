1   package br.com.codesolver.session;
2   
3   import java.io.Serializable;
4   import java.text.MessageFormat;
5   import java.util.Objects;
6   import java.util.concurrent.ConcurrentHashMap;
7   import java.util.concurrent.ConcurrentMap;
8   import java.util.logging.Logger;
9   
10  /**
11   * Os parâmetros da sessão devem possuir nomes únicos e tipos definidos.
12   *
13   * @author <a href="mailto:luciano@codesolver.com.br">Luciano Vieira Rodrigues</a>
14   * @since 2025-08-26
15   * @param <T> Tipo de valor para o parâmetro de sessão.
16   */
17  public class SessionParam<T> implements Serializable {
18  	
19  	/**
20  	 * Log da classe.
21  	 */
22  	private static final Logger LOGGER = Logger.getLogger(SessionParam.class.getName());
23  	
24  	/**
25  	 * Nomes dos parâmetros.
26  	 */
27  	private static final ConcurrentMap<String, Boolean> NAMES = new ConcurrentHashMap<String, Boolean>();
28  	
29  	/**
30  	 * Nome interno do parâmetro.
31  	 */
32  	private String name;
33  
34  	/**
35  	 * Gerenciamento de nomes de parâmetros.
36  	 *
37  	 * @param name Nome único do parâmetro.
38  	 */
39  	public SessionParam(String name) {
40  		if (name == null) {
41  			String message = "Parâmetro não pode ser nulo.";
42  			LOGGER.severe(message);
43  			throw new SessionRuntimeException(message);
44  		} 
45  		if (name.isEmpty()) {
46  			String message = "Parâmetro está vazio.";
47  			LOGGER.severe(message);
48  			throw new SessionRuntimeException(message);
49  		} 
50  		if (NAMES.putIfAbsent(name, Boolean.TRUE) != null) {
51  			String message = MessageFormat.format("Parâmetro {0} já está em uso.", name);
52  			LOGGER.severe(message);
53  			throw new SessionRuntimeException(message);
54  		}
55  		this.name = name;
56  	}
57  
58  	/**
59  	 * Nome de identificação do parâmetro de sessão.
60  	 *
61  	 * @return name Nome do parâmetro de Sessão.
62  	 */
63  	public String getName() {
64  		return name;
65  	}
66  
67  	@Override
68  	// CHECKSTYLE:OFF
69  	public int hashCode() {
70  		final int prime = 31;
71  		int result = 1;
72  		result = prime * result + ((name == null) ? 0 : name.hashCode());
73  		return result;
74  	}
75  	// CHECKSTYLE:ON
76  
77  	@Override
78  	public boolean equals(Object obj) {
79  		if (this == obj) {
80  			return true;
81  		}
82  		if (obj == null || getClass() != obj.getClass()) {
83  			return false;
84  		}
85  		SessionParam<?> other = (SessionParam<?>) obj;
86  		return Objects.equals(name, other.name);
87  	}
88  
89  	/**
90  	 * Verifica se o parâmetro foi atribuído corretamente.
91  	 *
92  	 * @param value Apenas o valor para verificação.
93  	 */
94  	public void validate(T value) {
95  		if (value == null) {
96  			throw new SessionRuntimeException("Parâmetro nulo");
97  		}
98  	}
99  }

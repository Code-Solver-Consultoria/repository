364         assertTrue(4 == octectos.length);
365         assertTrue(octectos[0] == (byte)0xFF);
366         assertTrue(octectos[1] == (byte)0xFF);
367         assertTrue(octectos[2] == (byte)0xFF);
368         assertTrue(octectos[3] == (byte)0xFF);
369 
370         octectos = StringUtil.encodeIPv4("0.0.0.0");
371         assertTrue(4 == octectos.length);
372         assertTrue(octectos[0] == (byte)0x00);
373         assertTrue(octectos[1] == (byte)0x00);
374         assertTrue(octectos[2] == (byte)0x00);
375         assertTrue(octectos[3] == (byte)0x00);
376 
377         octectos = StringUtil.encodeIPv4("170.170.170.170");
378         assertTrue(4 == octectos.length);
379         assertTrue(octectos[0] == (byte)0xAA);
380         assertTrue(octectos[1] == (byte)0xAA);

3   import java.nio.ByteBuffer;
4   import java.nio.charset.Charset;
5   import java.util.Date;
6   import java.util.Random;
7   import java.util.UUID;
8   import java.util.concurrent.atomic.AtomicLong;

10  /**
11   * Gerador de UUID tipo 1 (time based). É um número de 128 bits, conforme a estrutura:
12   *
13   * <table border="1">
14   *     <caption>Estrutura do UUID v1 (time-based)</caption>
15   *     <tr>
16   *       <td colspan="10">0</td>
17   *       <td colspan="10">1</td>
18   *       <td colspan="10">2</td>
19   *       <td colspan="2">3</td>
20   *     </tr>
21   *     <tr>
22   *       <td>0</td>
23   *       <td>1</td>
24   *       <td>2</td>
25   *       <td>3</td>
26   *       <td>4</td>
27   *       <td>5</td>
28   *       <td>6</td>
29   *       <td>7</td>
30   *       <td>8</td>
31   *       <td>9</td>
32   *       <td>0</td>
33   *       <td>1</td>
34   *       <td>2</td>
35   *       <td>3</td>
36   *       <td>4</td>
37   *       <td>5</td>
38   *       <td>6</td>
39   *       <td>7</td>
40   *       <td>8</td>
41   *       <td>9</td>
42   *       <td>0</td>
43   *       <td>1</td>
44   *       <td>2</td>
45   *       <td>3</td>
46   *       <td>4</td>
47   *       <td>5</td>
48   *       <td>6</td>
49   *       <td>7</td>
50   *       <td>8</td>
51   *     </tr>
52   *     <tr>
53   *       <td colspan="32" style="text-align:center;">time_low</td>
54   *     </tr>
55   *     <tr>
56   *       <td colspan="16" style="text-align:center;">time_mid</td>
57   *       <td colspan="16" style="text-align:center;">time_hi and version</td>
58   *     </tr>
59   *     <tr>
60   *       <td colspan="8" style="text-align:center;">clock_seq_hi</td>
61   *       <td colspan="8" style="text-align:center;">clock_seq_low</td>
62   *       <td colspan="16" style="text-align:center;">node (0-1)</td>
63   *     </tr>
64   *     <tr>
65   *       <td colspan="32" style="text-align:center;">node (2-5)</td>
66   *     </tr>
67   * </table>
68   *
69   * <ul>
70   * 	<li><b>time_low</b>: O menor campo de data e hora, com 4 bytes.</li>
71   * 	<li><b>time_mid</b>: O campo do meio da data e hora, com 6 bytes.</li>
72   * 	<li><b>time_hi</b>: O maior campo da data e hora, com 2 bytes.</li>
73   * 	<li><b>version</b>: São os 4 bits do campo time_hi multiplexado para o valor 1 (número 1).</li>
74   *	<li><b>clock_seq_hi</b>: Maior campo do sequencial do relógio multiplexado com a variante do algorítimo. É um byte.</li>
75   *	<li><b>clock_seq_low</b>: Menor campo do sequencial do relógio com 1 byte.</li>
76   *	<li><b>node</b>: Espaço para identificar a origem de criação do UUID.</li>
77   * </ul>
78   *
79   * <p>
80   * <i>Data e Hora é um número de 60 bits.</i>
81   *
82   * <b>Baseado no documento: http://www.ietf.org/rfc/rfc4122.txt</b>
83   *
84   * @author Luciano Vieira Rodrigues
85   * @version 1.0
86   * @since 04/04/2013
87   */
88  @SuppressWarnings("checkstyle:AbbreviationAsWordInName")
89  public class UUIDGenerator {
90  
91      /**
92       * Números randômicos para criação dos identificadores universais.
93       */
94      private static final Random RANDOM = new Random(System.currentTimeMillis());
95  
96      /**
97       * Tamanho máximo para o nome do gerador de UUID.
98       */
99      private static final int NODE_MAX_LENGHT = 6;
100 
101     /**
102      * A data base do Java é 01/01/1970, mas a data base do UTC e do cálculo do UUID é
103      * 15/10/1582. Para ajustar a data, é utilizado uma subtração dos nanosegundos.
104      */
105     private static final long TIME_OFF_SET = 0x01b21dd213814000L;
106 
107     /**
108      * Como a precisão do Java é em milisegundos, a multiplicação ajuda a criar uma
109      * data com precisão em nanosegundos, como o UTC.
110      */
111     private static final long TIME_MULTIPLIER = 10000L;
112 
113     /**
114      * O incremento para desempate de chaves quando são geradas no mesmo milisegundo.
115      * Para desempate, é possível criar 10000 no mesmo milisegundo.
116      */
117     private static final long TIME_ADVANCE = 100L;
118 
119     /**
120      * Número randômico para gerar a sequência do UUID.
121      */
122     private static final short CLOCK = (short) RANDOM.nextInt();
123 
124     /** Máscara para remover o último byte na formação do número MSB. */
125     private static final int MASK_REMOVE_LAST_BYTE = ~0xF000;
126 
127     /** Máscara para configuar o UUID como tipo 1. */
128     private static final int MASK_UUID_TYPE_1 = 0x1000;
129 
130     /**
131      * Último nanosegundo usado para gerar uma chave UUID.
132      */
133     private AtomicLong lastNanos;
134 
135     /**
136      * Nome de identificação do gerador de UUID.
137      */
138     private final String node;
139 
140     /**
141      * Construtor padrão.
142      *
143      * @param node Nome de identificação para o gerador de UUID. Não pode ser nulo, vazio ou possuir
144      *             mais que 6 caracteres.
145      * @throws UUIDGeneratorNodeInvalid Nome de identificação inválido.
146      */
147     public UUIDGenerator(String node) {
148         if (node == null || node.isEmpty() || node.length() > NODE_MAX_LENGHT) {
149             throw new UUIDGeneratorNodeInvalid();
150         }
151         this.node 		= node;
152         this.lastNanos 	= new AtomicLong(Long.MIN_VALUE);
153     }
154 
155     /**
156      * Gera uma chave Universal.
157      *
158      * @return UUID versão 1.
159      */
160     public UUID generate() {
161         return new UUID(makeMSB(), makeLSB());
162     }
163 
164     /**
165      * É o primeiro parâmetro para o construtor do UUID.
166      *
167      * <p>
168      * Possui os campos:
169      *
170      * <ul>
171      * 	<li>time_low</li>
172      * 	<li>time_mid</li>
173      *  <li>time_hi and version</li>
174      * </ul>
175      *
176      * @return long
177      */
178     private long makeMSB() {
179         long result = 0L;
180         long time = makeTime();
181         // Trocando a posição dos bytes.
182         int timeHigh	= (int) (time >>> Integer.SIZE);
183         int timeLow		= (int) time;
184         int timeMidHigh	= (timeHigh << Short.SIZE) | (timeHigh >>> Short.SIZE);
185         // Adicionando a versão...
186         timeMidHigh &= MASK_REMOVE_LAST_BYTE; 	// Remove o ultimo byte...
187         timeMidHigh |= MASK_UUID_TYPE_1; 		// Configura o UUID para o tipo 1 (time-based)
188         // Reconstruindo o número:
189         result = (long) timeMidHigh;
190         result = result << Integer.SIZE >>> Integer.SIZE; 		// Removendo o sinal...
191         result = ((long) timeLow) << Integer.SIZE | result;
192         return result;
193     }
194 
195     /**
196      * Cria um data no formato UTC, em nanosegundos.
197      *
198      * @return long
199      */
200     private long makeTime() {
201         long result = 0L;
202 
203         result = System.currentTimeMillis();
204         // Multiplicando para conseguir os nanosegundos.
205         result *= TIME_MULTIPLIER;
206         // Diferença dos nanosegundos para a data UTC (15/10/1582).
207         result += TIME_OFF_SET;
208 
209         // Se houve empate nos momento de geração de dois UUIDs, incrementa-se
210         // para gerar um valor único.
211         do {
212 	        if (result > lastNanos.get()) {
213 	        	lastNanos.set(result);
214 	        } else {
215 	        	result = lastNanos.addAndGet(TIME_ADVANCE);
216 	        }
217         } while (lastNanos.get() > result);
218 
219         return result;
220     }
221 
222     /**
223      * É o segundo parâmetro passado para o construtor do UUID.
224      *
225      * <p>
226      * Representando:
227      *
228      * <ul>
229      * 	<li>clock_seq_hi</li>
230      * 	<li>clock_seq_low</li>
231      * 	<li>node</li>
232      * </ul>
233      *
234      * @return long
235      */
236     @SuppressWarnings("checkstyle:MagicNumber")
237     private long makeLSB() {
238         long result			= 0L;
239         ByteBuffer buffer 	= ByteBuffer.allocate(Long.SIZE / Byte.SIZE);
240         buffer.putShort(CLOCK);
241         for (int i = 0; i < (buffer.limit() - (Short.SIZE / Byte.SIZE) - node.length()); i++) {
242             buffer.put((byte) RANDOM.nextInt());
243         }
244         buffer.put(node.getBytes(Charset.defaultCharset()));
245         buffer.flip();
246         result  = buffer.getLong();
247         result  = result << 2 >>> 2; 	    // remove dois bits MSB
248         result |= 2L << 62; 				// Configura os dois bits MSB para 10
249         return result;
250     }
251 
252     /**
253      * Recupera a data de geração.
254      *
255      * @param uuid UUID
256      * @return Date
257      */
258     public static Date getDate(UUID uuid) {
259         if (uuid.version() != 1) {
260             throw new UUIDGeneratorVersionInvalid();
261         }
262         long time = (uuid.timestamp() - TIME_OFF_SET) / TIME_MULTIPLIER;
263         return new Date(time);
264     }

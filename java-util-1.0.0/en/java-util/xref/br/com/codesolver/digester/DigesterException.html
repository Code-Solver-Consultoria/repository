1   package br.com.codesolver.digester;
2   
3   /**
4    * Informa um erro ocorrido em {@link DigesterSUN}.
5    *
6    * <p>
7    * Esta exceção fornece vários construtores para diferentes cenários de uso:
8    *
9    * <ul>
10   *   <li>Construtor sem argumentos para exceções genéricas.</li>
11   *   <li>Construtor com mensagem para descrição detalhada do erro.</li>
12   *   <li>Construtor com mensagem e causa para encadeamento de exceções.</li>
13   *   <li>Construtor com causa para encapsular exceções subjacentes.</li>
14   *   <li>Construtor protegido com opções avançadas para supressão e gravabilidade da pilha de chamadas.</li>
15   * </ul>
16   *
17   * @author <a href="mailto:luciano@codesolver.com.br">Luciano Vieira Rodrigues</a>
18   * @since 2025-08-26
19   */
20  public class DigesterException extends RuntimeException {
21  
22      /**
23       * Constrói uma nova {@code DigesterException} com {@code null} como mensagem de detalhe.
24       *
25       * <p>
26       * A causa não é inicializada e pode ser definida posteriormente através de {@link Throwable#initCause}.
27       */
28      public DigesterException() {
29      }
30  
31      /**
32       * Constrói uma nova {@code DigesterException} com a mensagem de detalhe especificada.
33       *
34       * @param message mensagem explicando o motivo da exceção
35       */
36      public DigesterException(String message) {
37          super(message);
38      }
39  
40      /**
41       * Constrói uma nova {@code DigesterException} com a mensagem de detalhe e causa especificadas.
42       *
43       * @param message mensagem de detalhe (recuperada posteriormente pelo método {@link #getMessage()})
44       * @param cause causa da exceção (recuperada posteriormente pelo método {@link #getCause()}).
45       *              O valor {@code null} indica que a causa é inexistente ou desconhecida.
46       */
47      public DigesterException(String message, Throwable cause) {
48          super(message, cause);
49      }
50  
51      /**
52       * Constrói uma nova {@code DigesterException} com a causa especificada.
53       *
54       * @param cause causa subjacente desta exceção
55       */
56      public DigesterException(Throwable cause) {
57          super(cause);
58      }
59  
60      /**
61       * Constrói uma nova {@code DigesterException} com mensagem de detalhe, causa,
62       * supressão e gravabilidade da pilha de chamadas configuráveis.
63       *
64       * @param message mensagem de detalhe (recuperada posteriormente pelo método {@link #getMessage()})
65       * @param cause causa da exceção (recuperada posteriormente pelo método {@link #getCause()}).
66       *              O valor {@code null} indica que a causa é inexistente ou desconhecida.
67       * @param enableSuppression indica se a supressão está habilitada ou desabilitada
68       * @param writableStackTrace indica se a pilha de chamadas pode ser gravada
69       */
70      protected DigesterException(String message, Throwable cause, boolean enableSuppression, boolean writableStackTrace) {
71          super(message, cause, enableSuppression, writableStackTrace);
72      }
73  }

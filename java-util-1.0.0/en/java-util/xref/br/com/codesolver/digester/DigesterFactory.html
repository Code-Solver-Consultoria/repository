1   package br.com.codesolver.digester;
2   
3   import java.security.NoSuchAlgorithmException;
4   import java.text.MessageFormat;
5   import java.util.logging.Logger;
6   
7   /**
8    * Cria o processador de HASH adequado para o algorítimo informado.
9    *
10   * @author <a href="mailto:luciano@codesolver.com.br">Luciano Vieira Rodrigues</a>
11   * @since 2025-08-26
12   * @see AlgorithmType
13   */
14  public final class DigesterFactory {
15  
16  	/** Log da classe. */
17  	private static final Logger LOGGER = Logger.getLogger(DigesterFactory.class.getName());
18  
19  	/** 
20  	 * Construtor oculto.
21  	 */
22  	private DigesterFactory() {
23  	}
24  
25  	/**
26  	 * Recupera o processador de HASH adequando para o algorítimo.
27  	 *
28  	 * @param <T> Um descendente de {@link Digester}.
29  	 * @param algorithm {@link AlgorithmType}.
30  	 * @return {@link Digester}.
31  	 * @throws DigesterException Algorítimo inválido.
32  	 */
33  	@SuppressWarnings("unchecked")
34  	public static <T extends Digester> T getInstance(AlgorithmType algorithm) {
35  		if (algorithm == null) {
36              String message = "Algoritimo não pode ser nulo.";
37              LOGGER.severe(message);
38              throw new DigesterException(message);
39          }
40  		T result = null;
41  		try {
42  			switch (algorithm) {
43  				case CRC16: 
44  					result = (T) new DigesterCRC16();
45  					break;
46  				default:
47  					result = (T) new DigesterSUN(algorithm);
48  			}
49  		} catch (NoSuchAlgorithmException e) {
50  			String message = MessageFormat.format("Algoritimo {0} inválido.", algorithm.name());
51              LOGGER.severe(message);
52              throw new DigesterException(message, e);
53  		} 
54  		return result;
55  	}
56  }

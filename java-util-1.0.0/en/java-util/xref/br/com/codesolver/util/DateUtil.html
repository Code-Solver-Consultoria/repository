80      /**
81       * Remove o horário de uma data.
82       *
83       * @param date {@link Date}.
84       * @return {@link Date}.
85       */
86      public static Date truncate(Date date) {
87          Date result = null;
88          if (date != null) {
89              Calendar calendar = new GregorianCalendar();
90              calendar.setTime(date);
91              calendar.set(Calendar.HOUR, 0);
92              calendar.set(Calendar.MINUTE, 0);
93              calendar.set(Calendar.SECOND, 0);
94              calendar.set(Calendar.MILLISECOND, 0);
95              calendar.set(Calendar.AM_PM, Calendar.AM);
96              result = calendar.getTime();
97          }
98          return result;
99      }

230     public static boolean isWeekend(Date date) {
231         GregorianCalendar calendar = new GregorianCalendar();
232         calendar.setTime(date);
233         return calendar.get(Calendar.DAY_OF_WEEK) == Calendar.SATURDAY
234                 || calendar.get(Calendar.DAY_OF_WEEK) == Calendar.SUNDAY;
235     }

237     /**
238      * Calcula a diferença de milisegundos entre duas datas, onde a data inicial
239      * deve ser menor que a data final.
240      *
241      * @param begin  {@link Date} inicial.
242      * @param finish {@link Date} final.
243      * @return Diferença em milisegundos entre as datas.
244      */
245     public static long milisBetween(Date begin, Date finish) {
246         long result = 0;
247         if (begin.before(finish)) {
248             result = finish.getTime() - begin.getTime();
249         }
250         return result;
251     }

253     /**
254      * Calcula a diferença de segundos entre duas datas, onde a data inicial
255      * deve ser menor que a data final.
256      *
257      * @param begin  {@link Date} inicial.
258      * @param finish {@link Date} final.
259      * @return Diferença em segundos entre as datas.
260      */
261     public static long secondsBetween(Date begin, Date finish) {
262         Calendar cbegin     = new GregorianCalendar();
263         Calendar cfinish    = new GregorianCalendar();
264 
265         cbegin.setTime(begin);
266         cbegin.set(Calendar.MILLISECOND, 0);
267         cfinish.setTime(finish);
268         cfinish.set(Calendar.MILLISECOND, 0);
269 
270         long difference = milisBetween(cbegin.getTime(), cfinish.getTime());
271         long result = 0;
272         if (difference >= SECOND_IN_MILIS) {
273             result = difference / SECOND_IN_MILIS;
274         }
275         return result;
276     }

278     /**
279      * Calcula a diferença de minutos entre duas datas, onde a data inicial
280      * deve ser menor que a data final.
281      *
282      * @param begin  {@link Date} inicial.
283      * @param finish {@link Date} final.
284      * @return Diferença em minutos entre as datas.
285      */
286     public static long minutesBetween(Date begin, Date finish) {
287         long difference = secondsBetween(begin, finish);
288         long result = 0;
289         if (difference >= MINUTE_IN_SECONDS) {
290             result = difference / MINUTE_IN_SECONDS;
291         }
292         return result;
293     }
